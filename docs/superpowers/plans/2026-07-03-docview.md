# docview Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a local live doc viewer (Astro + Starlight) that lets Claude Code users see project docs, git activity, and checkbox progress at a glance, with `astro dev` running as the runtime product.

**Architecture:** A single Astro project inside `docview/` folder of any target repo. Astro Content Collections with a glob loader read md files from user-configured paths. Custom Astro pages compose Starlight's docs UX with a dashboard (Live Activity banner + 2-column: recent commits timeline / category progress cards). `chokidar` watches the target repo for changes, invalidates scan caches, and triggers Vite HMR to update the browser. Client-side polling every 5s hits `/api/activity.json` for live status.

**Tech Stack:** Node 18+, Astro 4.x, Starlight 0.30+, TypeScript 5, chokidar 3, zod 3, Vitest 1.x, Playwright 1.40+, `@astrojs/mdx`, `rehype-mermaid`.

## Global Constraints

- **Node version floor:** `>=18.0.0`
- **Astro major:** `^4` (glob loader API required)
- **TypeScript strict mode ON** everywhere
- **All checkboxes in md files use `- [ ]` / `- [x]`** (no other bullet syntax)
- **Frontmatter status enum:** exactly `draft | in-progress | done`
- **Category rule matching:** first match wins; fallback = target-repo-root-relative path's first segment (root files → `Root`)
- **Live activity polling interval:** 5000 ms (constant; not configurable in MVP)
- **Astro dev port:** 4321 (auto-fallback if busy)
- **No production build / deploy in MVP** — `astro dev` IS the product
- **Commit prefixes:** `feat` | `fix` | `test` | `docs` | `chore` | `refactor` | `build`
- **TDD Iron Law:** no production code without a failing test first (exception: pure scaffolding tasks noted below)
- **Every task ends with a commit.** No batched multi-task commits.

export type CategoryRule = {
  name: string;
  match: string | string[];
  icon?: string;
};

export type DocviewConfig = {
  include: string[];
  exclude: string[];
  categories: CategoryRule[];
};

export const DEFAULT_CONFIG: DocviewConfig = {
  include: [
    'docs/**/*.md',
    'AGENTS.md',
    'README.md',
    'CLAUDE.md',
    '.omc/**/*.md',
    '.claude/plans/*.md',
  ],
  exclude: [
    '**/node_modules/**',
    'docview/**',
    '.git/**',
  ],
  categories: [
    { name: 'ADR',   match: 'docs/adr/**',                                icon: '📐' },
    { name: 'PRD',   match: ['docs/prd*', '.omc/prd-*'],                  icon: '📋' },
    { name: 'Plans', match: '.claude/plans/**',                           icon: '🗺' },
    { name: 'Root',  match: ['AGENTS.md', 'README.md', 'CLAUDE.md'],      icon: '📄' },
  ],
};

export function mergeConfig(override: Partial<DocviewConfig>): DocviewConfig {
  const include = override.include ?? DEFAULT_CONFIG.include;
  const exclude = override.exclude ?? DEFAULT_CONFIG.exclude;
  const overriddenNames = new Set((override.categories ?? []).map((c) => c.name));
  const kept = DEFAULT_CONFIG.categories.filter((c) => !overriddenNames.has(c.name));
  const categories = [...kept, ...(override.categories ?? [])];
  return { include, exclude, categories };
}

export async function loadConfig(rootDir: string = process.cwd()): Promise<DocviewConfig> {
  const path = await import('node:path');
  const fs = await import('node:fs/promises');
  const configPath = path.join(rootDir, 'docview.config.ts');
  try {
    await fs.access(configPath);
  } catch {
    return DEFAULT_CONFIG;
  }
  const mod = await import(/* @vite-ignore */ configPath);
  const user = (mod.default ?? mod.config ?? {}) as Partial<DocviewConfig>;
  return mergeConfig(user);
}
```

```ts
import type { CategoryRule } from './config';

function globToRegex(glob: string): RegExp {
  // Minimal micromatch-lite: **, *, ?
  let re = '^';
  let i = 0;
  while (i < glob.length) {
    const c = glob[i];
    if (c === '*' && glob[i + 1] === '*') {
      re += '.*';
      i += 2;
      if (glob[i] === '/') i++; // consume the following separator so `a/**/b` matches `a/b`
    } else if (c === '*') {
      re += '[^/]*';
      i++;
    } else if (c === '?') {
      re += '[^/]';
      i++;
    } else if ('.^$+(){}|\\'.includes(c)) {
      re += '\\' + c;
      i++;
    } else {
      re += c;
      i++;
    }
  }
  re += '$';
  return new RegExp(re);
}

function ruleMatches(path: string, rule: CategoryRule): boolean {
  const patterns = Array.isArray(rule.match) ? rule.match : [rule.match];
  return patterns.some((p) => globToRegex(p).test(path));
}

export function resolveCategory(path: string, rules: CategoryRule[]): string {
  for (const rule of rules) {
    if (ruleMatches(path, rule)) return rule.name;
  }
  const firstSeg = path.split('/')[0];
  return firstSeg === path ? 'Root' : firstSeg;
}

const CHECKBOX_LINE = /^\s*[-*+]\s+\[([ xX])\]\s+/;

export function scanCheckboxes(markdown: string): { total: number; checked: number } {
  let total = 0;
  let checked = 0;
  let inFence = false;
  const lines = markdown.replace(/\r\n/g, '\n').split('\n');
  for (const raw of lines) {
    const line = raw.replace(/`[^`\n]*`/g, ''); // strip inline code
    if (/^\s*```/.test(line)) {
      inFence = !inFence;
      continue;
    }
    if (inFence) continue;
    const m = CHECKBOX_LINE.exec(line);
    if (m) {
      total++;
      if (m[1] === 'x' || m[1] === 'X') checked++;
    }
  }
  return { total, checked };
}
```

import { execFile as execFileCb } from 'node:child_process';
import { promisify } from 'node:util';
const execFile = promisify(execFileCb);

export type Commit = {
  sha: string;
  author: string;
  relTime: string;
  subject: string;
  files: string[];
};

export async function scanGitLog(rootDir: string, limit = 50): Promise<Commit[]> {
  const SEP = '|~docview~|';
  const FMT = `%H${SEP}%an${SEP}%ar${SEP}%s`;
  const { stdout } = await execFile(
    'git',
    ['-C', rootDir, 'log', `--pretty=format:${FMT}`, '--name-only', '-z', `-n${limit}`],
    { maxBuffer: 32 * 1024 * 1024 },
  );
  const commits: Commit[] = [];
  const chunks = stdout.split('\0\0').filter(Boolean);
  for (const chunk of chunks) {
    const [meta, ...fileParts] = chunk.split('\n');
    const [sha, author, relTime, subject] = meta.split(SEP);
    const files = (fileParts.join('\n')).split('\0').map((f) => f.trim()).filter(Boolean);
    commits.push({ sha, author, relTime, subject, files });
  }
  return commits;
}

export type DirtyEntry = { path: string; status: 'M' | 'A' | 'D' | '??' | 'R' };

export async function scanDirtyFiles(rootDir: string): Promise<DirtyEntry[]> {
  const { stdout } = await execFile('git', ['-C', rootDir, 'status', '--porcelain', '-z'], {
    maxBuffer: 8 * 1024 * 1024,
  });
  const entries: DirtyEntry[] = [];
  const parts = stdout.split('\0').filter(Boolean);
  for (const p of parts) {
    const code = p.slice(0, 2).trim();
    const rel = p.slice(3);
    const status = (code[0] === '?' ? '??' : (code[0] as DirtyEntry['status'])) ?? 'M';
    entries.push({ path: rel, status });
  }
  return entries;
}

```ts
import path from 'node:path';
import fs from 'node:fs/promises';
import fg from 'fast-glob';
import { scanDirtyFiles, type DirtyEntry } from './scan';
import type { DocviewConfig } from './config';

export type Activity = {
  dirty: DirtyEntry[];
  recentlyModified: Array<{ path: string; mtime: number }>;
  activePlan: { path: string; title: string } | null;
  generatedAt: number;
};

const FIVE_MIN = 5 * 60 * 1000;
const THIRTY_MIN = 30 * 60 * 1000;

async function firstHeading(absPath: string): Promise<string> {
  try {
    const content = await fs.readFile(absPath, 'utf8');
    const m = content.match(/^#\s+(.+)$/m);
    return m ? m[1].trim() : path.basename(absPath, path.extname(absPath));
  } catch {
    return path.basename(absPath);
  }
}

async function collectMatches(rootDir: string, cfg: DocviewConfig): Promise<Array<{ path: string; mtime: number; abs: string }>> {
  const abs = await fg(cfg.include, { cwd: rootDir, absolute: true, ignore: cfg.exclude });
  const stats = await Promise.all(
    abs.map(async (a) => {
      const s = await fs.stat(a).catch(() => null);
      return s ? { path: path.relative(rootDir, a).split(path.sep).join('/'), mtime: s.mtimeMs, abs: a } : null;
    }),
  );
  return stats.filter((x): x is NonNullable<typeof x> => Boolean(x));
}

export async function getActivity(rootDir: string, cfg: DocviewConfig): Promise<Activity> {
  const now = Date.now();
  const [dirty, matches] = await Promise.all([
    scanDirtyFiles(rootDir).catch(() => []),
    collectMatches(rootDir, cfg),
  ]);

  const recentlyModified = matches
    .filter((m) => now - m.mtime <= FIVE_MIN)
    .sort((a, b) => b.mtime - a.mtime)
    .slice(0, 10)
    .map(({ path, mtime }) => ({ path, mtime }));

  const planCandidates = matches
    .filter((m) => m.path.startsWith('.claude/plans/') && now - m.mtime <= THIRTY_MIN)
    .sort((a, b) => b.mtime - a.mtime);
  const fallbackCandidates = matches
    .filter((m) => now - m.mtime <= THIRTY_MIN)
    .sort((a, b) => b.mtime - a.mtime);
  const chosen = planCandidates[0] ?? fallbackCandidates[0] ?? null;
  const activePlan = chosen ? { path: chosen.path, title: await firstHeading(chosen.abs) } : null;

  return { dirty, recentlyModified, activePlan, generatedAt: now };
}

import chokidar from 'chokidar';
import path from 'node:path';
import type { DocviewConfig } from './config';

export function startWatcher(rootDir: string, cfg: DocviewConfig, onChange: (relPath: string) => void) {
  const watchPaths = [
    ...cfg.include.map((g) => path.join(rootDir, g)),
    path.join(rootDir, '.git/HEAD'),
    path.join(rootDir, '.git/index'),
  ];
  const watcher = chokidar.watch(watchPaths, {
    ignored: cfg.exclude,
    ignoreInitial: true,
    followSymlinks: false,
    awaitWriteFinish: { stabilityThreshold: 200, pollInterval: 50 },
  });
  const handler = (absPath: string) => {
    const rel = path.relative(rootDir, absPath).split(path.sep).join('/');
    onChange(rel);
  };
  watcher.on('add', handler).on('change', handler).on('unlink', handler);
  return {
    close: () => watcher.close(),
  };
}

### Task 15: FilterPanel component + Starlight sidebar override

**Files:**

- Create: `src/components/Sidebar.astro` (Starlight sidebar override)
- Modify: `astro.config.mjs` (`starlight` `components: { Sidebar: '/src/components/Sidebar.astro' }`)

**Interfaces:**
- Behaviour: filter panel with three groups (status / category / tags). URL query string + LocalStorage sync. Client-side only.

- [ ] **Step 1: Write `src/components/FilterPanel.astro`**

```astro
---
import { getCollection } from 'astro:content';
const docs = await getCollection('docs');
const statuses = ['draft', 'in-progress', 'done'];
const categoriesSet = new Set<string>();
const tagsSet = new Set<string>();
for (const d of docs) {
  if (d.data.category) categoriesSet.add(d.data.category);
  for (const t of d.data.tags ?? []) tagsSet.add(t);
}
const categories = [...categoriesSet].sort();
const tags = [...tagsSet].sort();
---
<div id="docview-filter">
  <h3 style="margin:8px 0 4px;">Status</h3>
  {statuses.map((s) => (
    <label style="display:block;"><input type="checkbox" data-kind="status" value={s} /> {s}</label>
  ))}
  <h3 style="margin:8px 0 4px;">Category</h3>
  {categories.map((c) => (
    <label style="display:block;"><input type="checkbox" data-kind="category" value={c} /> {c}</label>
  ))}
  <h3 style="margin:8px 0 4px;">Tags</h3>
  {tags.map((t) => (
    <label style="display:block;"><input type="checkbox" data-kind="tag" value={t} /> {t}</label>
  ))}
</div>
<script>
  const KEY = 'docview.filter';
  const root = document.getElementById('docview-filter');
  if (root) {
    const saved = JSON.parse(localStorage.getItem(KEY) || '{}');
    for (const input of root.querySelectorAll('input')) {
      const k = input.dataset.kind, v = input.value;
      if ((saved[k] || []).includes(v)) input.checked = true;
      input.addEventListener('change', () => {
        const state = {};
        for (const kind of ['status','category','tag']) {
          state[kind] = [...root.querySelectorAll(`input[data-kind="${kind}"]:checked`)].map((i)=>i.value);
        }
        localStorage.setItem(KEY, JSON.stringify(state));
        const url = new URL(location.href);
        for (const kind of ['status','category','tag']) {
          if (state[kind].length) url.searchParams.set(kind, state[kind].join(','));
          else url.searchParams.delete(kind);
        }
        history.replaceState({}, '', url);
        // Emit a custom event so doc lists can re-filter client-side.
        window.dispatchEvent(new CustomEvent('docview:filter', { detail: state }));
      });
    }
  }
</script>
```
